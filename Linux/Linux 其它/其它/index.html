    \$() `` ${} $[] $(()) [] (()) [[]]

    $()     # 命令替换，同 `` ，$() 不是所有的 shell 都支持

    ``      # 命令替换，基本可在所有 unix shell 中使用，移植性较高

    ${}     # 变量替换，比较精确地界定变量名称的范围
    变量设定方式 说明
    ${变量#关键字}
    ${变量##关键字}    		 若变量内容从头开始的数据符合 "关键字"，则将符合的最短数据删除
    若变量内容从头开始的数据符合 "关键字"，则将符合的最常数据删除
    ${变量%关键字}
    ${变量%%关键字}    		 若变量内容从后向前的数据符合 "关键字"，则将符合的最短数据删除
    若变量内容从后向前的数据符合 "关键字"，则将符合的最常数据删除
    ${变量/旧字符串/新字符串}
    ${变量//旧字符串/新字符串}    		 若变量内容符合 "旧字符串"，则 "第一个旧字符串会被新字符串取代"
    若变量内容符合 "旧字符串"，则 "所有旧字符串会被新字符串取代"    		 
    $[]     # 进行数学运算，bash 只能作整数运算，对于浮点数会当做字符串处理

    $(())   # 进行数学运算，bash 只能作整数运算，对于浮点数会当做字符串处理

    []      # 条件测试

    [[]]    # 条件测试，[] 的加强版

    (())    # [] 的加强版，除标准的数学运算符外，还增加了：
var++       # 后自增
var--       # 后自减
++var       # 前自增
--var       # 前自减
!           # 逻辑取反
~           # 位取反
**          # 幂运算
<<          # 左位移
>>          # 右位移
&           # 位布尔与
|           # 位布尔或
&&          # 逻辑与
||          # 逻辑或

    启动级别

    # 说明：CentOS 8开始不再使用/etc/inittab文件进行默认的启动级别配置，而使用比sysvinit的运行级别更为自由的target替代
# 第3级别用 multi-user.target 替代
# 第5级别用 graphial.target 替代

    # 1、在目录/etc/rc.d/init.d下有许多服务器脚本程序，一般称为服务(service)
# 2、在/etc/rc.d下有7个名为rcN.d的目录，对应系统的7个运行级别
# 3、rcN.d目录下都是一些符号链接文件，这些链接文件都指向init.d目录下的service脚本文件，命名规则为K+nn+服务名或S+nn+服务名，其中nn为两位数字。
# 4、系统会根据指定的运行级别进入对应的rcN.d目录，并按照文件名顺序检索目录下的链接文件
     对于以K开头的文件，系统将终止对应的服务
     对于以S开头的文件，系统将启动对应的服务
# 5、查看运行级别用：runlevel
# 6、进入其它运行级别用：init N
# 7、另外init0为关机，init 6为重启系统
    init级别 systemd目标 作用
    0 poweroff.target 系统关机模式，系统默认启动级别不能设置为0
    1 rescue.target 单用户模式，root权限，用户系统维护，禁止远程登录，像Windows的安全模式
    2 multi-user.target 多用户模式，没有NFS和网络支持
    3 multi-user.target 完整多用户字符模式，有NFS和网络支持
    4 multi-user.target 系统未使用，保留一般不用
    5 graphical.target 图形化模式，登陆进入图形GUI模式
    6 reboot.target 重启模式，系统默认启动级别不能设置为6 
    init 0      # 关机
init 6      # 重启
init 3      # 进入字符界面
init 5      # 进入图形界面

    systemctl set-default multi-user.target     # 设置第3级别为默认启动级别
systemctl isolate graphical.target      # 设置第5级别为默认启动级别
systemctl get-default       # 查看当前默认启动级别

    启动原理

    CentOS 6

    1、加载BIOS，若硬盘为第一引导
2、读取硬盘MBR中的 boot loader 即 GRUB
    MBR硬盘的0柱面、0磁头、1扇区称为主引导扇区 < 也叫主引导记录 >
    它由主引导程序、硬盘分区表 DPT < Disk Partition table > 和硬盘有效标志 < 55AA > 组成
    硬盘默认一个扇区占512字节
        主引导程序：占446个字节，负责从活动分区中装载，并运行系统引导程序
        分区表：占64个字节，硬盘中分区有多少以及每一分区的大小都记在其中，每个分区表项长16字节，
               为分区项1、分区项2、分区项3、分区项4，只存4个分区表
        有效标示位：占2个字节，固定为55AA，如果这个标志位0x55AA，就认为这个是MBR
3、根据boot loader的设定，到引导分区记在Kernel，Kernel会开始侦测硬件并加载驱动程序
4、驱动加载后，Kernel会主动执行init程序，而init会去到run-level信息
5、init 执行/etc/rc.d/rc.sysinit 文件来准备软件执行的作业环境 < 如：网络、时区等 >
6、init 执行 run-level 下各个服务并启动 < script 方式 >
7、init 执行开机后自动运行脚本 /etc/rc.d/rc.local 文件
8、init 执行虚拟终端机控制程序 mingetty 来启动 login 程序，最后等待用户登录

    1、BIOS加电自检，根据设置读取第一个启动设备

    2.2 由于系统数据都会存储到硬盘中，所以BIOS会指定启动的设备以便读取磁盘中的操作系统内核文件，但由于不同而系统的文件格式不同，因此会以启动引导程序来处理内核文件的加载 < LOAD > 所以这个启动引导程序就被称为 boot loader，会被放置到启动设备的第一个扇区中

    4.1、systemd 执行 sysyinit.target 初始化系统，basic.target 准备操作系统
    4.2、启动 multi-user.target
    4.3、systemd 执行 multi-user.target 下的 /etc/rc.d/rc.local 文件
    4.4、systemd 执行 multi-user.target 下的 getty.target 以及登录服务
    4.5、systemd 执行 graphical 需要的服务
5、在内核加载完毕、进行硬件检测和驱动程序的加载后，内核会调用第一个程序，就是systemd，使用pstree命令可以看到 systemd 的PID为   1，systemd 会准备软件运行的环境，包括系统的主机名、网络、语言、文件系统格式及其他服务的启动。而所有的操作都会通过systemd的   默认启动服务集合/etc/systemd/system/default.target < 实际上这是个软链接，而真正链接到的文件会随着默认启动级别的变化而变化 >

    Free 中的 buff/cache

    参考：https://mp.weixin.qq.com/s/lNYdslDNpOi9pYvMQx5reQ

    buff（Buffer Cache）是一种 I/O 缓存，用于内存和硬盘的缓冲，是io设备的读写缓冲区。

cache（Page Cache）是一种高速缓存，用于 CPU 和内存之间的缓冲 ,是文件系统的 cache。把读取过的数据保存起来，重新读取时若命中（找到需要的数据）就不要去读硬盘了，若没有命中就读硬盘。其中的数据会根据读取频率进行组织，把最频繁读取的内容放在最容易找到的位置，把不再读的内容不断往后排，直至从中删除。

# 简单说，buff 是即将要被写入磁盘的，cache 是被从磁盘中读出来的。

    # 清除 buff/cache

sync && sync && sync && sleep 10
echo 1 > /proc/sys/vm/drop_caches       # 清除page cache
echo 2 > /proc/sys/vm/drop_caches       # 清除回收slab分配器中的对象（包括目录项缓存和inode缓存）
echo 3 > /proc/sys/vm/drop_caches       # 清除pagecache和slab分配器中的缓存对象。

    OpenStack 的日志格式如下：

<时间戳><日志等级><代码模块><Request ID><日志内容><源代码位置>

时间戳          # 日志记录的时间，包括 年 月 日 时 分 秒 毫秒
日志等级        # INFO WARNING ERROR DEBUG等
代码模块        # 当前运行的模块
Request ID     # 日志会记录连续不同的操作，为了便于区分和增加可读性，每个操作都被分配唯一的Request ID,便于查找日志内容 这是日志的主体，记录当前正在执行的操作和结果等重要信息
源代码位置       # 日志代码的位置，包括方法名称，源代码文件的目录位置和行号。这一项不是所有日志都有
